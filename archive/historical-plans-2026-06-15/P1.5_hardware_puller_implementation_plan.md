## 1. 概述

### 1.1 目标

实现 Hardware Puller 状态机，替代当前 `handle_pushbuffer_submit_batch()` 的简单 switch 循环。

### 1.2 当前状态 vs 目标状态

| 组件 | 当前 (Skeleton) | 目标 (ADR-021) |
|------|----------------|----------------|
| `doorbell_emu.h/cpp` | 27行，仅计数 | 完整 doorbell 寄存器仿真 |
| `hardware_puller_emu.h/cpp` | IDLE/PROCESSING 2状态 | 6状态完整状态机 |
| `scheduler/` | 空目录 | Global Scheduler FIFO |
| `handle_pushbuffer_submit_batch` | 同步处理 | doorbell 触发异步路径 |

### 1.3 状态机定义 (ADR-021)

```
┌───────────┐
│  IDLE     │ ← 等待 doorbell write
└─────┬─────┘
      │ (hal_doorbell_read() 检测到写入)
      ▼
┌───────────┐
│ FETCH     │ ← 通过 HAL 从设备内存 DMA 读取 GPFIFO entry
└─────┬─────┘
      │ (读取完成)
      ▼
┌───────────┐
│ DECODE    │ ← 解析 method + payload + semaphore 字段
└─────┬─────┘
      │
 ┌────┴────┐
 ▼         ▼
SCHEDULE  SEMAPHORE ← GPU_OP_FENCE: WAIT/RELEASE
   │          │ (sem wait 完成)
   ▼          │
DISPATCH     │ ← 路由到正确的执行引擎
   │          │
   └────┬────┘
        ▼
┌───────────┐
│ COMPLETE  │ ← release==1 → 触发中断
└─────┬─────┘
      │
      ▼
┌───────────┐
│  NEXT     │ ← 继续下个 entry（Puller 缓存了多个 entry）
└─────┬─────┘
      │ (所有 entry 处理完)
      ▼
┌───────────┐
│  IDLE     │ ← 等待下一个 doorbell
└───────────┘
```

### 2.1 扩展 doorbell_emu (sim/hardware/doorbell_emu.h/cpp)

**目标**: 实现完整的 doorbell 寄存器仿真，支持 HAL 触发

#### doorbell_emu.h

```cpp
#pragma once

#include <cstdint>
#include <array>
#include <functional>
#include "gpu_types.h"

class DoorbellEmu {
 public:
  static constexpr u64 BASE   = 0x00001000;  // ADR-021 §6
  static constexpr u64 STRIDE = 0x00000040;  // 每 queue 间隔
  static constexpr u32 MAX_QUEUES = 32;

  using DoorbellCallback = std::function<void(u32 queue_id)>;

  DoorbellEmu() : counts_(), pending_() {
    counts_.fill(0);
    pending_.fill(false);
  }

  // CPU 通过 PCIe write 写入 doorbell (HAL 调用)
  void write(u32 queue_id) {
    if (queue_id >= MAX_QUEUES) return;
    counts_[queue_id]++;
    pending_[queue_id] = true;
    // 触发 Puller 回调
    if (callback_) {
      callback_(queue_id);
    }
  }

  // Puller 轮询检测
  bool poll(u32 queue_id) const {
    if (queue_id >= MAX_QUEUES) return false;
    return pending_[queue_id];
  }

  // Puller 消费 doorbell (清除 pending)
  void acknowledge(u32 queue_id) {
    if (queue_id < MAX_QUEUES) {
      pending_[queue_id] = false;
    }
  }

  // 获取累积 ring 次数 (调试用)
  u64 getRingCount(u32 queue_id) const {
    if (queue_id >= MAX_QUEUES) return 0;
    return counts_[queue_id];
  }

  // 注册 doorbell 回调 (Puller 监听)
  void setCallback(DoorbellCallback cb) { callback_ = std::move(cb); }

 private:
  std::array<u64, MAX_QUEUES> counts_;
  std::array<bool, MAX_QUEUES> pending_;   // 是否有待消费的 doorbell
  DoorbellCallback callback_;              // Puller 通知回调
};
```

### 2.2 重写 hardware_puller_emu (sim/hardware/hardware_puller_emu.h/cpp)

**目标**: 实现 ADR-021 定义的 6 状态完整状态机

#### hardware_puller_emu.h

```cpp
#pragma once

#include <cstdint>
#include <vector>
#include <atomic>
#include <thread>
#include <mutex>
#include <condition_variable>
#include "gpu_types.h"
#include "hal/gpu_hal.h"
#include "hardware/doorbell_emu.h"

struct gpu_gpfifo_entry;

class HardwarePullerEmu {
 public:
  // 状态枚举 (ADR-021 §1)
  enum class State {
    IDLE,      // 等待 doorbell
    FETCH,     // DMA 读取 GPFIFO entry
    DECODE,    // 解析 method/payload/semaphore
    SCHEDULE,  // 路由到执行引擎
    DISPATCH,  // 执行引擎处理中
    COMPLETE,   // entry 完成，等待 NEXT
    SEMAPHORE  // 等待 semaphore (WAIT opcode)
  };

  HardwarePullerEmu(struct gpu_hal_ops* hal,
                    DoorbellEmu* doorbell,
                    void* scheduler);
  ~HardwarePullerEmu();

  // Puller 线程入口 (后台运行)
  void start();
  void stop();

  // Doorbell 触发 (从 doorbell_emu 调用)
  void onDoorbell(u32 queue_id);

  // 查询状态 (调试用)
  State currentState() const { return state_; }
  const char* stateName() const;

  // 通过 HAL 读取设备内存中的 GPFIFO entry
  // 由 handle_pushbuffer_submit_batch 调用，将 entry 写入模拟设备内存后触发 doorbell
  void submitBatch(u64 gpfifo_gpu_addr, u32 entry_count);

 private:
  void runLoop();           // Puller 主循环
  bool fetchEntry();        // FETCH 状态
  void decodeEntry();       // DECODE 状态
  void handleSchedule();    // SCHEDULE 状态
  void handleDispatch();    // DISPATCH 状态
  void handleComplete();    // COMPLETE 状态
  void handleSemaphore();   // SEMAPHORE 状态
  void transitionTo(State next);

  // Semaphore 操作 (ADR-021 §3)
  bool waitSemaphore(u64 semaphore_va, u32 expected_value);
  void releaseSemaphore(u64 semaphore_va, u32 release_value);

  // 引擎路由 (ADR-021 §5)
  enum class EngineType { COMPUTE, COPY, FIRMWARE };
  EngineType getEngineType(u32 method) const;
  void dispatchToEngine(EngineType engine, const gpu_gpfifo_entry& entry);

  struct gpu_hal_ops* hal_;
  DoorbellEmu* doorbell_;
  void* scheduler_;  // GlobalScheduler*

  State state_;
  std::thread thread_;
  std::mutex mutex_;
  std::condition_variable cv_;
  std::atomic<bool> running_{false};

  // 当前处理的 entry
  gpu_gpfifo_entry current_entry_;
  u64 current_gpfifo_addr_;
  size_t current_index_;
  size_t total_entries_;

  // Semaphore 等待队列
  struct SemaphoreWait {
    u64 semaphore_va;
    u32 expected_value;
    std::atomic<bool>* signaled;
  };
  std::vector<SemaphoreWait> pending_waits_;
};
```

#### hardware_puller_emu.cpp

```cpp
#include "hardware/hardware_puller_emu.h"
#include "scheduler/global_scheduler.h"

#include <iostream>
#include <cstring>

HardwarePullerEmu::HardwarePullerEmu(struct gpu_hal_ops* hal,
                                     DoorbellEmu* doorbell,
                                     void* scheduler)
    : hal_(hal),
      doorbell_(doorbell),
      scheduler_(scheduler),
      state_(State::IDLE),
      current_index_(0),
      total_entries_(0) {
  // 注册 doorbell 回调
  doorbell_->setCallback([this](u32 qid) { onDoorbell(qid); });
}

HardwarePullerEmu::~HardwarePullerEmu() { stop(); }

void HardwarePullerEmu::start() {
  std::lock_guard<std::mutex> lock(mutex_);
  if (running_.load()) return;
  running_.store(true);
  thread_ = std::thread(&HardwarePullerEmu::runLoop, this);
}

void HardwarePullerEmu::stop() {
  {
    std::lock_guard<std::mutex> lock(mutex_);
    if (!running_.load()) return;
    running_.store(false);
    cv_.notify_all();
  }
  if (thread_.joinable()) {
    thread_.join();
  }
}

void HardwarePullerEmu::onDoorbell(u32 queue_id) {
  (void)queue_id;
  std::lock_guard<std::mutex> lock(mutex_);
  if (state_ == State::IDLE) {
    cv_.notify_one();  // 唤醒 Puller
  }
}

void HardwarePullerEmu::runLoop() {
  while (running_.load()) {
    State local_state;
    {
      std::unique_lock<std::mutex> lock(mutex_);
      cv_.wait(lock, [this] {
        return !running_.load() ||
               (state_ == State::IDLE && doorbell_->poll(0));
      });
      if (!running_.load()) break;
      local_state = state_;
    }

    switch (local_state) {
      case State::IDLE:
        // 检查 doorbell
        if (doorbell_->poll(0)) {
          doorbell_->acknowledge(0);
          transitionTo(State::FETCH);
        }
        break;

      case State::FETCH:
        if (!fetchEntry()) {
          transitionTo(State::IDLE);
        }
        break;

      case State::DECODE:
        decodeEntry();
        break;

      case State::SCHEDULE:
        handleSchedule();
        break;

      case State::DISPATCH:
        handleDispatch();
        break;

      case State::SEMAPHORE:
        handleSemaphore();
        break;

      case State::COMPLETE:
        handleComplete();
        break;
    }
  }
}

bool HardwarePullerEmu::fetchEntry() {
  if (current_index_ >= total_entries_) {
    return false;  // 全部处理完毕
  }

  // 通过 HAL mem_read 从设备内存读取 entry
  u64 entry_addr = current_gpfifo_addr_ + current_index_ * sizeof(gpu_gpfifo_entry);
  int ret = hal_->mem_read(hal_->ctx, entry_addr,
                            &current_entry_, sizeof(gpu_gpfifo_entry));
  if (ret != 0 || !current_entry_.valid) {
    std::cerr << "[HardwarePuller] FETCH failed at index " << current_index_ << "\n";
    return false;
  }

  std::cout << "[HardwarePuller] FETCH entry[" << current_index_
            << "] method=0x" << std::hex << current_entry_.method << std::dec << "\n";
  return true;
}

void HardwarePullerEmu::decodeEntry() {
  // 解析 semaphore 字段 (ADR-021 §3)
  if (current_entry_.semaphore_va != 0) {
    std::cout << "[HardwarePuller] DECODE: sem_va=0x" << std::hex
              << current_entry_.semaphore_va << " value=" << current_entry_.semaphore_value
              << " release=" << current_entry_.release << std::dec << "\n";
  }
  // 根据 method 字段决定下一步
  transitionTo(State::SCHEDULE);
}

void HardwarePullerEmu::handleSchedule() {
  EngineType engine = getEngineType(current_entry_.method);

  if (engine == EngineType::FIRMWARE && current_entry_.method == GPU_OP_FENCE) {
    // 检查 WAIT vs RELEASE
    if (current_entry_.release == 0) {
      transitionTo(State::SEMAPHORE);
      return;
    }
  }

  transitionTo(State::DISPATCH);
}

void HardwarePullerEmu::handleDispatch() {
  EngineType engine = getEngineType(current_entry_.method);
  dispatchToEngine(engine, current_entry_);
  transitionTo(State::COMPLETE);
}

void HardwarePullerEmu::handleSemaphore() {
  if (current_entry_.release == 0) {
    // WAIT: 阻塞等待 semaphore
    if (waitSemaphore(current_entry_.semaphore_va, current_entry_.semaphore_value)) {
      transitionTo(State::COMPLETE);
    }
    // 未完成则继续等待 (下次循环检查)
  } else {
    // RELEASE: 写入 semaphore
    releaseSemaphore(current_entry_.semaphore_va, current_entry_.semaphore_value);
    transitionTo(State::COMPLETE);
  }
}

void HardwarePullerEmu::handleComplete() {
  // 中断触发 (ADR-021 §4)
  if (current_entry_.release == 1) {
    std::cout << "[HardwarePuller] COMPLETE: triggering interrupt\n";
    hal_->interrupt_raise(hal_->ctx, 0);  // MSI-X vector 0
  }

  current_index_++;
  if (current_index_ < total_entries_) {
    transitionTo(State::FETCH);  // 继续处理下个 entry
  } else {
    transitionTo(State::IDLE);  // 全部完成
  }
}

void HardwarePullerEmu::transitionTo(State next) {
  std::lock_guard<std::mutex> lock(mutex_);
  state_ = next;
}

HardwarePullerEmu::EngineType HardwarePullerEmu::getEngineType(u32 method) const {
  switch (method) {
    case GPU_OP_LAUNCH_KERNEL:
    case GPU_OP_LAUNCH_CPU_TASK:
      return EngineType::COMPUTE;
    case GPU_OP_MEMCPY:
    case GPU_OP_MEMSET:
      return EngineType::COPY;
    case GPU_OP_FENCE:
      return EngineType::FIRMWARE;
    default:
      return EngineType::COMPUTE;
  }
}

void HardwarePullerEmu::dispatchToEngine(EngineType engine,
                                          const gpu_gpfifo_entry& entry) {
  switch (engine) {
    case EngineType::COMPUTE:
      std::cout << "[HardwarePuller] DISPATCH to COMPUTE\n";
      // TODO: 调用 GlobalScheduler
      break;
    case EngineType::COPY:
      std::cout << "[HardwarePuller] DISPATCH to COPY\n";
      // TODO: 调用 GlobalScheduler
      break;
    case EngineType::FIRMWARE:
      std::cout << "[HardwarePuller] DISPATCH to FIRMWARE\n";
      // TODO: 调用 GlobalScheduler
      break;
  }
}

bool HardwarePullerEmu::waitSemaphore(u64 semaphore_va, u32 expected_value) {
  // Phase 1: 简单轮询 (ADR-021 §3)
  // Phase 2: 改为事件驱动
  for (int i = 0; i < 1000; i++) {
    u32 current_value = 0;
    int ret = hal_->mem_read(hal_->ctx, semaphore_va, &current_value, sizeof(current_value));
    if (ret == 0 && current_value >= expected_value) {
      return true;
    }
    std::this_thread::sleep_for(std::chrono::microseconds(1));
  }
  return false;  // 超时
}

void HardwarePullerEmu::releaseSemaphore(u64 semaphore_va, u32 release_value) {
  hal_->mem_write(hal_->ctx, semaphore_va, &release_value, sizeof(release_value));
  std::cout << "[HardwarePuller] RELEASE semaphore at 0x" << std::hex << semaphore_va
            << " = " << std::dec << release_value << "\n";
}

const char* HardwarePullerEmu::stateName() const {
  switch (state_) {
    case State::IDLE:      return "IDLE";
    case State::FETCH:      return "FETCH";
    case State::DECODE:     return "DECODE";
    case State::SCHEDULE:   return "SCHEDULE";
    case State::DISPATCH:   return "DISPATCH";
    case State::SEMAPHORE:  return "SEMAPHORE";
    case State::COMPLETE:   return "COMPLETE";
    default:               return "UNKNOWN";
  }
}
```

### 2.3 新增 Global Scheduler (sim/scheduler/global_scheduler.h/cpp)

**目标**: 实现简单的 FIFO 调度器，路由到不同引擎 (ADR-021 §5)

#### global_scheduler.h

```cpp
#pragma once

#include <queue>
#include <mutex>
#include <functional>
#include "gpu_types.h"

class HardwarePullerEmu;  // 前向声明

struct WorkItem {
  enum class Type { KERNEL, MEMCPY, MEMSET, FENCE, CPU_TASK };
  Type type;
  gpu_gpfifo_entry entry;
  u64 fence_id;  // 完成时写入

  static WorkItem fromEntry(const gpu_gpfifo_entry& e) {
    WorkItem item;
    item.entry = e;
    item.fence_id = 0;
    // 根据 method 推断 type
    switch (e.method) {
      case GPU_OP_LAUNCH_KERNEL:   item.type = Type::KERNEL; break;
      case GPU_OP_MEMCPY:          item.type = Type::MEMCPY; break;
      case GPU_OP_MEMSET:          item.type = Type::MEMSET; break;
      case GPU_OP_FENCE:           item.type = Type::FENCE; break;
      case GPU_OP_LAUNCH_CPU_TASK: item.type = Type::CPU_TASK; break;
      default:                     item.type = Type::KERNEL; break;
    }
    return item;
  }
};

class GlobalScheduler {
 public:
  using ComputeHandler = std::function<void(const gpu_gpfifo_entry&)>;
  using CopyHandler = std::function<void(const gpu_gpfifo_entry&)>;
  using FirmwareHandler = std::function<void(const gpu_gpfifo_entry&)>;

  GlobalScheduler();

  // 注册引擎处理函数 (由仿真层设置)
  void setComputeHandler(ComputeHandler h) { compute_handler_ = std::move(h); }
  void setCopyHandler(CopyHandler h) { copy_handler_ = std::move(h); }
  void setFirmwareHandler(FirmwareHandler h) { firmware_handler_ = std::move(h); }

  // Puller 调用 submit() 提交工作项
  void submit(WorkItem item);

  // 执行调度 (FIFO)
  void schedule();

  // 查询状态
  size_t queueSize() const;
  bool empty() const { return queue_.empty(); }

 private:
  std::queue<WorkItem> queue_;
  mutable std::mutex mutex_;

  ComputeHandler compute_handler_;
  CopyHandler copy_handler_;
  FirmwareHandler firmware_handler_;
};
```

#### global_scheduler.cpp

```cpp

#include "hardware/hardware_puller_emu.h"

#include <iostream>

GlobalScheduler::GlobalScheduler() = default;

void GlobalScheduler::submit(WorkItem item) {
  std::lock_guard<std::mutex> lock(mutex_);
  queue_.push(std::move(item));
}

void GlobalScheduler::schedule() {
  while (!empty()) {
    WorkItem item;
    {
      std::lock_guard<std::mutex> lock(mutex_);
      if (queue_.empty()) return;
      item = queue_.front();
      queue_.pop();
    }

    switch (item.type) {
      case WorkItem::Type::KERNEL:
        if (compute_handler_) {
          compute_handler_(item.entry);
        }
        break;
      case WorkItem::Type::MEMCPY:
      case WorkItem::Type::MEMSET:
        if (copy_handler_) {
          copy_handler_(item.entry);
        }
        break;
      case WorkItem::Type::FENCE:
      case WorkItem::Type::CPU_TASK:
        if (firmware_handler_) {
          firmware_handler_(item.entry);
        }
        break;
    }

    // 写入 fence (如果已分配)
    if (item.fence_id != 0) {
      // hal_fence_signal(item.fence_id);
    }
  }
}

size_t GlobalScheduler::queueSize() const {
  std::lock_guard<std::mutex> lock(mutex_);
  return queue_.size();
}
```

### 2.4 改造 handle_pushbuffer_submit_batch

**目标**: 从同步处理改为 doorbell 触发异步路径

#### 修改 drv/gpgpu_device.cpp

```cpp
long GpgpuDevice::handlePushbufferSubmitBatch(void* argp) {
  auto* args = static_cast<struct gpu_pushbuffer_args*>(argp);
  if (!args || args->count == 0) {
    return -EFAULT;
  }

  // Phase 1: 走 doorbell 路径 (ADR-021 §2)
  // 1. 分配设备内存 (通过 HAL)
  u64 gpfifo_gpu_addr = 0;
  size_t gpfifo_size = args->count * sizeof(gpu_gpfifo_entry);
  int ret = hal_mem_alloc(hal_, gpfifo_size, &gpfifo_gpu_addr);
  if (ret != 0 || gpfifo_gpu_addr == 0) {
    std::cerr << "[GpgpuDevice] PUSHBUFFER: hal_mem_alloc failed\n";
    return -ENOMEM;
  }

  // 2. 将 entries 写入设备内存 (通过 HAL)
  ret = hal_mem_write(hal_, gpfifo_gpu_addr, (void*)args->entries_addr, gpfifo_size);
  if (ret != 0) {
    std::cerr << "[GpgpuDevice] PUSHBUFFER: hal_mem_write failed\n";
    hal_mem_free(hal_, gpfifo_gpu_addr);
    return -EFAULT;
  }

  // 3. 预分配 fence_id (ADR-021 §5)
  u64 fence_id = 0;
  hal_fence_create(hal_, &fence_id);
  args->fence_id = fence_id;

  // 4. 触发 doorbell (通过 HAL doorbell_ring)
  // doorbell_->write(args->stream_id);  // stream_id 作为 queue_id

  std::cout << "[GpgpuDevice] PUSHBUFFER: submitted " << args->count
            << " entries to queue " << args->stream_id << "\n";
  return 0;
}
```

**注意**: 完整实现需要 `hardware_puller_emu_` 成员和 `doorbell_emu_` 成员注入到 `GpgpuDevice`。

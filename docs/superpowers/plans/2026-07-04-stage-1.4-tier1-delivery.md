namespace {
constexpr unsigned long INVALID_PFN = ~0UL;

struct SimPageMigration {
    unsigned long device_mem_size;
    std::vector<unsigned char> device_memory;
    std::map<unsigned long, bool> page_on_device;
    std::map<unsigned long, unsigned long> page_table;  // offset → pfn
    int migration_count = 0;
};
} // anonymous namespace

extern "C" {

// 新增 API
unsigned long sim_pm_lookup_pfn(struct sim_page_migration *pm, unsigned long offset) {
    if (!pm) return INVALID_PFN;
    auto *p = reinterpret_cast<SimPageMigration *>(pm);
    auto it = p->page_table.find(offset);
    if (it == p->page_table.end()) return INVALID_PFN;
    return it->second;
}

int sim_pm_migrate_to_device(struct sim_page_migration *pm,
                              unsigned long offset, const void *src, unsigned long size) {
    if (!pm || !src) return -EINVAL;
    auto *p = reinterpret_cast<SimPageMigration *>(pm);
    if (offset + size > p->device_mem_size) return -EFAULT;

    memcpy(p->device_memory.data() + offset, src, size);
    p->page_on_device[offset] = true;
    // 新增：注册 pfn
    p->page_table[offset] = offset / 4096;  // simple model: offset/PAGE_SIZE = pfn
    p->migration_count++;
    return 0;
}

int sim_pm_migrate_to_system(struct sim_page_migration *pm,
                              unsigned long offset, void *dst, unsigned long size) {
    if (!pm || !dst) return -EINVAL;
    auto *p = reinterpret_cast<SimPageMigration *>(pm);
    if (offset + size > p->device_mem_size) return -EFAULT;

    memcpy(dst, p->device_memory.data() + offset, size);
    p->page_on_device[offset] = false;
    // 新增：清除 pfn
    p->page_table.erase(offset);
    p->migration_count++;
    return 0;
}

} // extern "C"
```

namespace {

// New: fault cause enum (Linux VM_FAULT_* aligned subset)
constexpr int SIM_FAULT_CAUSE_READ  = 0;  // VM_FAULT_READ
constexpr int SIM_FAULT_CAUSE_WRITE = 1;  // VM_FAULT_WRITE

struct SimPageFaultHandler {
    struct mm_struct *mm;
    int fault_count = 0;
    unsigned long last_fault_addr = 0;
    int last_fault_cause = SIM_FAULT_CAUSE_READ;
};

} // anonymous namespace

extern "C" {

// 兼容旧 API
void sim_pfh_inject_fault(struct sim_page_fault_handler *pfh,
                           unsigned long addr, unsigned long *pfn_out) {
    sim_pfh_inject_fault_with_cause(pfh, addr, pfn_out, SIM_FAULT_CAUSE_READ);
}

// 新增 API
void sim_pfh_inject_fault_with_cause(struct sim_page_fault_handler *pfh,
                                      unsigned long addr, unsigned long *pfn_out,
                                      int cause) {
    if (!pfh) return;
    auto *p = reinterpret_cast<SimPageFaultHandler *>(pfh);
    p->fault_count++;
    p->last_fault_addr = addr;
    p->last_fault_cause = cause;
    if (pfn_out) *pfn_out = addr / 4096;
}

int sim_pfh_get_last_fault_cause(struct sim_page_fault_handler *pfh) {
    if (!pfh) return SIM_FAULT_CAUSE_READ;
    return reinterpret_cast<SimPageFaultHandler *>(pfh)->last_fault_cause;
}

} // extern "C"
```

#include "kfd_sim_bridge.h"

#include "plugins/gpu_driver/sim/page_migration.h"
#include "plugins/gpu_driver/sim/page_fault_handler.h"

#include <map>
#include <mutex>

namespace {

// 内部 sim 状态（单例，全局唯一）
struct KfdSimState {
    struct sim_page_migration *pm = nullptr;
    std::map<u64, u64> gpu_va_to_pfn;  // gpu_va → pfn
    u32 page_count = 0;
    u64 next_gpu_va = 0x100000ULL;  // 基地址
};

KfdSimState g_state;
std::mutex g_state_mutex;

constexpr u64 GPU_VA_STRIDE = 0x1000ULL;  // 4KB per page

} // anonymous namespace

extern "C" {

void kfd_sim_reset(void) {
    std::lock_guard<std::mutex> lock(g_state_mutex);
    if (g_state.pm) sim_pm_destroy(g_state.pm);
    g_state.pm = sim_pm_create(16 * 1024 * 1024);  // 16 MB device memory
    g_state.gpu_va_to_pfn.clear();
    g_state.page_count = 0;
    g_state.next_gpu_va = 0x100000ULL;
}

u64 kfd_sim_lookup_pfn(u64 gpu_va) {
    std::lock_guard<std::mutex> lock(g_state_mutex);
    auto it = g_state.gpu_va_to_pfn.find(gpu_va);
    if (it == g_state.gpu_va_to_pfn.end()) return INVALID_PFN;
    return it->second;
}

u32 kfd_sim_get_page_count(void) {
    std::lock_guard<std::mutex> lock(g_state_mutex);
    return g_state.page_count;
}

long kfd_sim_handle_map_memory(struct gpu_map_memory_args *args) {
    if (!args) return -EFAULT;
    if (args->size == 0) return -EINVAL;
    if (args->n_devices == 0 || args->n_devices > 8) return -EINVAL;
    if (!g_state.pm) return -ENODEV;

    std::lock_guard<std::mutex> lock(g_state_mutex);

    // 真实调用 sim 原语：migrate device memory → sim
    char dummy_src[4096] = {0};
    u64 offset = (g_state.page_count * 4096) % sim_pm_get_device_mem_size(g_state.pm);
    int ret = sim_pm_migrate_to_device(g_state.pm, offset, dummy_src, args->size);
    if (ret < 0) return ret;

    // 分配 gpu_va 并记录映射
    u64 gpu_va = g_state.next_gpu_va;
    g_state.next_gpu_va += GPU_VA_STRIDE;
    g_state.gpu_va_to_pfn[gpu_va] = sim_pm_lookup_pfn(g_state.pm, offset);
    g_state.page_count++;

    args->gpu_va = gpu_va;
    args->n_success = args->n_devices;
    return 0;
}

long kfd_sim_handle_unmap_memory(struct gpu_unmap_memory_args *args) {
    if (!args) return -EFAULT;
    if (args->n_devices == 0 || args->n_devices > 8) return -EINVAL;
    if (!g_state.pm) return -ENODEV;

    std::lock_guard<std::mutex> lock(g_state_mutex);

    // 查找并清除 gpu_va 映射
    auto it = g_state.gpu_va_to_pfn.find(/* gpu_va lookup by handle - simplified */);
    if (it != g_state.gpu_va_to_pfn.end()) {
        g_state.gpu_va_to_pfn.erase(it);
        g_state.page_count--;
    }

    args->n_success = args->n_devices;
    return 0;
}

long kfd_sim_handle_get_process_aperture(struct gpu_get_process_aperture_args *args) {
    if (!args) return -EFAULT;
    if (args->num_nodes == 0 || args->num_nodes > 8) return -EINVAL;
    if (args->apertures_ptr == 0) return -EFAULT;

    // Tier-1 范围内：从 sim 状态推导 aperture info
    std::lock_guard<std::mutex> lock(g_state_mutex);
    u64 device_mem_size = g_state.pm ? sim_pm_get_device_mem_size(g_state.pm) : 0;

    struct gpu_aperture_info apertures[8] = {};
    for (u32 i = 0; i < args->num_nodes; i++) {
        apertures[i].gpu_id = i;
        apertures[i].lds_base = 0x200000ULL;
        apertures[i].lds_limit = 0x200000ULL + 0x10000ULL;
        apertures[i].scratch_base = 0x300000ULL;
        apertures[i].scratch_limit = 0x300000ULL + 0x100000ULL;
        apertures[i].gpuvm_base = 0x100000ULL;
        apertures[i].gpuvm_limit = 0x100000ULL + device_mem_size;
    }

    // 复制到用户态（userspace pointer）
    if (copy_to_user(args->apertures_ptr, apertures,
                     args->num_nodes * sizeof(struct gpu_aperture_info)) != 0) {
        return -EFAULT;
    }
    return 0;
}

long kfd_sim_handle_update_queue(struct gpu_update_queue_args *args) {
    if (!args) return -EFAULT;
    if (args->queue_handle == 0) return -EINVAL;
    if (args->queue_flags & ~0xFu) return -EINVAL;

    // Tier-1 范围内：验证 queue handle 在 va_space 中有效
    // （简化版：仅 log + return 0，va_space 集成留待 B.2.5）
    return 0;
}

} // extern "C"
